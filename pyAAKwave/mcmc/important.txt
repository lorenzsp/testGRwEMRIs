Here it is an example of how I did it
mkdir /u/lsperi/conda-envs/scalar_few_env/lib/python3.7/site-packages/few-1.4.0-py3.7-linux-x86_64.egg/few/files
cp /u/lsperi/testGRwEMRIs/pyAAKwave/ModifiedFEW/few/files/FluxesEdot_scalar_tensor.dat  /u/lsperi/conda-envs/scalar_few_env/lib/python3.7/site-packages/few-1.4.0-py3.7-linux-x86_64.egg/few/files/
